int Solution::solve(vector<int> &A) {
    int root = INT_MIN;
    stack<int> st;
    for(int i=0;i<A.size();i++){
        while(st.size() && A[i]>st.top()){
            root = st.top();
            st.pop();
        }
        if(root>A[i]) return false;
        st.push(A[i]);
    }
    return true;
}
