bool isPrime(int num){
    if(num==1) return false;
    for(int i=2;i*i<=num;i++){
        if(num%i==0) return false;
    }
    return true;
}
vector<int> Solution::primesum(int A) {
    int num = A;
    vector<int> vec(2);
    for(int i=2;i<=num;i++){
        if(isPrime(i))
        if(isPrime(num-i)){
            vec[0]=i;
            vec[1]=num-i;
            break;
        }
    }
    return vec;
}
