vector<vector<int> > Solution::subsets(vector<int> &A) {
    sort(A.begin(),A.end());
    vector<vector<int>> res = {{}};
    for(int i=0;i<A.size();i++){
        int n=res.size();
        for(int j=0;j<n;j++){
            vector<int> temp = res[j];
            temp.push_back(A[i]);
            res.push_back(temp);
        }
    }
    sort(res.begin(),res.end());
    return res;
    
    
}
