int Solution::isPalindrome(string s) {
    int ans = 1;
    int n=s.size();
    int i=0,j=n-1;
    while(i<j){
        while(!isalpha(s[i]) && !isdigit(s[i])){
            i++;
            if(i==n) break;
        }
        while(!isalpha(s[j]) && !isdigit(s[j])){
            j--;
            if(j==-1) break;
        }
        if(i>=j) break;
        s[i] = tolower(s[i]);
        s[j] = tolower(s[j]);
        if(s[i] != s[j]) return 0;
        i++,j--;
    }
    return 1;
}
